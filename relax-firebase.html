🔒 Firestore Security Rules
انسخ في Firebase Console → Firestore → Rules واستبدل ADMIN_UID
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    match /listings/{id} {
      allow read: if true;
      allow create: if request.auth != null
        && request.resource.data.userId == request.auth.uid;
      allow update, delete: if request.auth != null
        && (resource.data.userId == request.auth.uid
            || request.auth.uid == 'ADMIN_UID');
      match /reviews/{rid} {
        allow read: if true;
        allow create: if request.auth != null;
        allow delete: if request.auth.uid == 'ADMIN_UID';
      }
    }
    match /reports/{id} {
      allow create: if request.auth != null;
      allow read, delete: if request.auth.uid == 'ADMIN_UID';
    }
    match /bookings/{id} {
      allow create: if request.auth != null
        && request.resource.data.userId == request.auth.uid;
      allow read: if request.auth != null
        && (resource.data.userId == request.auth.uid
            || resource.data.adOwnerId == request.auth.uid
            || request.auth.uid == 'ADMIN_UID');
    }
    match /chats/{chatId}/messages/{msgId} {
      allow read, write: if request.auth != null;
    }
    match /messages_unread/{id} {
      allow create: if request.auth != null;
      allow read, delete: if request.auth != null
        && (resource.data.toUid == request.auth.uid
            || request.auth.uid == 'ADMIN_UID');
    }
    match /ratings/{id} {
      allow read: if true;
      allow create: if request.auth != null
        && request.resource.data.userId == request.auth.uid;
      allow delete: if request.auth.uid == 'ADMIN_UID';
    }
  }
}

📜 الشروط والخصوصية

شروط الاستخدام

يلتزم المستخدمون بعدم نشر محتوى مخالف للقانون أو الأخلاق. يحق للإدارة حذف أي إعلان يخالف الشروط. الموقع غير مسؤول عن صحة المعلومات المنشورة.

سياسة الخصوصية

نحافظ على خصوصية بياناتك. لا نبيع معلوماتك لأطراف ثالثة. نستخدم Firebase لتخزين البيانات بشكل آمن. يمكنك حذف حسابك في أي وقت.

ملفات الارتباط (Cookies)

نستخدم cookies لتحسين تجربة التصفح وحفظ إعداداتك مثل اللغة المفضلة.

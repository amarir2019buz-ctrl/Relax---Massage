🔒 Firestore Security Rules
انسخ في Firebase Console → Firestore → Rules واستبدل ADMIN_UID
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // ===== LISTINGS (الإعلانات) =====
    match /listings/{id} {
      allow read: if true;
      allow create: if request.auth != null
        && request.resource.data.userId == request.auth.uid;
      allow update: if request.auth != null
        && (resource.data.userId == request.auth.uid
            || request.auth.uid == 'ADMIN_UID');
      allow delete: if request.auth != null
        && (resource.data.userId == request.auth.uid
            || request.auth.uid == 'ADMIN_UID');

      // Reviews on listings
      match /reviews/{rid} {
        allow read: if true;
        allow create: if request.auth != null;
        allow update: if request.auth != null
          && request.resource.data.keys().hasOnly(['ownerReply','replyAt'])
          && resource.data.adOwnerId == request.auth.uid;
        allow delete: if request.auth.uid == 'ADMIN_UID';
      }
    }

    // ===== REPORTS (البلاغات) =====
    match /reports/{id} {
      allow create: if request.auth != null;
      allow read, delete: if request.auth.uid == 'ADMIN_UID';
    }

    // ===== BOOKINGS (الحجوزات) =====
    match /bookings/{id} {
      allow create: if request.auth != null
        && request.resource.data.userId == request.auth.uid;
      allow read: if request.auth != null
        && (resource.data.userId == request.auth.uid
            || resource.data.adOwnerId == request.auth.uid
            || request.auth.uid == 'ADMIN_UID');
      allow update: if request.auth != null
        && resource.data.adOwnerId == request.auth.uid;
      allow delete: if request.auth.uid == 'ADMIN_UID';
    }

    // ===== BOOKING NOTIFICATIONS =====
    match /booking_notifications/{id} {
      allow create: if request.auth != null;
      allow read: if request.auth != null
        && (resource.data.toUid == request.auth.uid
            || request.auth.uid == 'ADMIN_UID');
      allow update, delete: if request.auth != null
        && resource.data.toUid == request.auth.uid;
    }

    // ===== CHATS (المحادثات) =====
    match /chats/{chatId}/messages/{msgId} {
      allow read, write: if request.auth != null;
    }
    match /chats/{chatId}/meta/{doc} {
      allow read, write: if request.auth != null;
    }

    // ===== MESSAGES UNREAD (الرسائل غير المقروءة) =====
    match /messages_unread/{id} {
      allow create: if request.auth != null;
      allow read, delete: if request.auth != null
        && (resource.data.toUid == request.auth.uid
            || resource.data.fromUid == request.auth.uid
            || request.auth.uid == 'ADMIN_UID');
    }

    // ===== RATINGS (التقييمات) =====
    match /ratings/{id} {
      allow read: if true;
      allow create: if request.auth != null
        && request.resource.data.userId == request.auth.uid;
      allow delete: if request.auth.uid == 'ADMIN_UID';
    }

    // ===== PAYMENT REQUESTS (طلبات الدفع) =====
    match /payment_requests/{id} {
      allow create: if request.auth != null
        && request.resource.data.userId == request.auth.uid;
      allow read, update: if request.auth.uid == 'ADMIN_UID';
    }

    // ===== SITE REVIEWS (تقييمات الموقع) =====
    match /site_reviews/{id} {
      allow read: if resource.data.approved == true;
      allow create: if request.auth != null
        && request.resource.data.userId == request.auth.uid;
      allow update, delete: if request.auth.uid == 'ADMIN_UID';
    }

    // ===== FOLLOWS (المتابعة) =====
    match /follows/{id} {
      allow read: if true;
      allow create: if request.auth != null
        && request.resource.data.fromUid == request.auth.uid;
      allow delete: if request.auth != null
        && resource.data.fromUid == request.auth.uid;
    }

    // ===== USERS (المستخدمون) =====
    match /users/{uid} {
      allow read: if true;
      allow write: if request.auth != null
        && request.auth.uid == uid;
      allow update: if request.auth.uid == 'ADMIN_UID';
    }

    // ===== NOTIFICATIONS (الإشعارات) =====
    match /notifications/{id} {
      allow read: if request.auth != null
        && resource.data.toUid == request.auth.uid;
      allow create: if request.auth != null;
      allow update: if request.auth != null
        && resource.data.toUid == request.auth.uid;
      allow delete: if request.auth.uid == 'ADMIN_UID';
    }

    // ===== BLOG (المدونة - اختياري إذا أردت حفظ المقالات) =====
    match /blog/{id} {
      allow read: if true;
      allow write: if request.auth.uid == 'ADMIN_UID';
    }

  }
}

🔒 سياسة الخصوصية

آخر تحديث: مارس 2025

1. المعلومات التي نجمعها

نجمع المعلومات التي تقدمها عند إنشاء حساب (الاسم، البريد الإلكتروني، الصورة الشخصية)، والمعلومات المتعلقة بإعلاناتك (رقم الهاتف، المدينة، الأسعار)، وبيانات الاستخدام (الصفحات التي تزورها، وقت الزيارة).

2. كيف نستخدم معلوماتك

نستخدم بياناتك لتشغيل الموقع وتحسين تجربتك، وإرسال إشعارات متعلقة بإعلاناتك وحجوزاتك، وتحليل أداء الموقع عبر Google Analytics.

3. مشاركة البيانات

لا نبيع معلوماتك الشخصية لأطراف ثالثة. نشارك فقط المعلومات الضرورية مع Firebase (Google) لتخزين البيانات، وGoogle Analytics لتحليل الزيارات.

4. ملفات الارتباط (Cookies)

نستخدم Cookies لحفظ إعداداتك (اللغة، الوضع الليلي، المفضلة) وتحسين تجربة التصفح. يمكنك تعطيلها من إعدادات متصفحك.

5. حقوقك

يحق لك الوصول إلى بياناتك وتعديلها وحذفها في أي وقت من خلال ملفك الشخصي. لحذف حسابك نهائياً، تواصل معنا عبر WhatsApp.

6. أمان البيانات

نستخدم Firebase Authentication وFirestore المشفرَين من Google لحماية بياناتك. جميع الاتصالات مشفرة بـ HTTPS.

7. التواصل
